cmake_minimum_required(VERSION 3.18)
project(GameOfLife LANGUAGES CXX CUDA)

# Set C++ and CUDA standards
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CUDA_STANDARD 17)
set(CMAKE_CUDA_STANDARD_REQUIRED ON)

# Set CUDA architecture for RTX 4060 (Ada Lovelace - compute capability 8.9)
# Also include 8.6 for broader compatibility
set(CMAKE_CUDA_ARCHITECTURES 86 89)

# CUDA flags for optimization
set(CMAKE_CUDA_FLAGS "${CMAKE_CUDA_FLAGS} -O3 -use_fast_math")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /O2")

# Include CUDA headers for C++ files
include_directories(${CMAKE_CUDA_TOOLKIT_INCLUDE_DIRECTORIES})

# Find OpenCV
# Support both vcpkg and manual installation
if(NOT DEFINED OpenCV_DIR AND EXISTS "C:/opencv/build")
    set(OpenCV_DIR "C:/opencv/build/x64/vc16/lib")
endif()
find_package(OpenCV REQUIRED)
include_directories(${OpenCV_INCLUDE_DIRS})

message(STATUS "OpenCV found!")
message(STATUS "OpenCV version: ${OpenCV_VERSION}")
message(STATUS "OpenCV libraries: ${OpenCV_LIBS}")
message(STATUS "OpenCV include dirs: ${OpenCV_INCLUDE_DIRS}")

# Real-time visualization executable (optimized for performance)
add_executable(gameoflife_realtime
    main_realtime.cpp
    gameoflife.cu
)

# Link OpenCV libraries
target_link_libraries(gameoflife_realtime ${OpenCV_LIBS})

# Set CUDA properties
set_property(TARGET gameoflife_realtime PROPERTY CUDA_SEPARABLE_COMPILATION ON)

# Platform-specific settings
if(WIN32)
    set_property(TARGET gameoflife_realtime PROPERTY CUDA_RESOLVE_DEVICE_SYMBOLS ON)
endif()

# Print configuration info
message(STATUS "CUDA architectures: ${CMAKE_CUDA_ARCHITECTURES}")
message(STATUS "Real-time visualization executable: gameoflife_realtime")
message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
